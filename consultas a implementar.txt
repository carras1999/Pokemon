//1- Select de pokemon de cada usuario.

SELECT Usuarios.Nombre AS Usuario, Pokedex.NumPokemon AS NumPokemon, Pokedex.NomPokemon AS Pokemon, Pokedex.Tipo, Pokedex.Especie
FROM Pokedex INNER JOIN (Usuarios INNER JOIN PokeUsuarios ON Usuarios.Id = PokeUsuarios.IdUsuario) ON Pokedex.NumPokemon = PokeUsuarios.IdPokemon 
WHERE Usuarios.Nombre = "$user" /*esta variable esta sujeta a posibles cambios*/
ORDER BY Usuarios.Nombre, Pokedex.NumPokemon ASC;
